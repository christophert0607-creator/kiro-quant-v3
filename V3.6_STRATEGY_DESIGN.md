## 第一部分：資金管理 (根據第六章)

### 1.1 Kelly Formula 實現

```python
class KellyPositionSizer:
    """
    Kelly Formula: f* = (bp - q) / b
    - f*: 最佳投資比例
    - b: 淨賠率 (淨odds)
    - p: 贏的概率
    - q: 輸的概率 = 1 - p
    """
    
    def __init__(self, use_half_kelly=True):
        self.use_half_kelly = use_half_kelly
        self.kelly_factor = 0.5 if use_half_kelly else 1.0  # 半 Kelly 更穩定
        
    def calculate_position_size(self, win_rate, avg_win, avg_loss, portfolio_value):
        """
        計算倉位大小
        
        參數:
        - win_rate: 歷史贏率 (0.0 - 1.0)
        - avg_win: 平均獲利金額
        - avg_loss: 平均虧損金額 (正數)
        - portfolio_value: 組合總值
        
        返回:
        - 建議投入金額
        """
        # 計算淨賠率 b
        b = avg_win / avg_loss
        
        # 計算概率
        p = win_rate
        q = 1 - p
        
        # Kelly Formula
        if b <= 0:
            return 0
            
        kelly = (b * p - q) / b
        
        # 應用 Kelly Factor (半 Kelly)
        kelly = kelly * self.kelly_factor
        
        # 限制在合理範圍內 (0% - 20%)
        kelly = max(0, min(kelly, 0.20))
        
        # 計算實際金額
        position_value = portfolio_value * kelly
        
        return position_value
```

```python
class TransactionCostCalculator:
    """
    交易成本計算器
    根據書籍建議：真實交易成本必須計入回測
    """
    
    def __init__(self):
        # 佣金 (根據不同券商)
        self.commission_rate = {
            'US_STOCK': 0.001,    # 0.1% (約 $0.005/股, 最低 $1)
            'HK_STOCK': 0.001,    # 0.1% + 印花稅 0.1%
            'futures': 0.0002,    # 期貨較低
        }
        
        # 滑點模型 (根據流動性)
        self.slippage_model = {
            'large_cap': 0.0005,   # 大型股 0.05%
            'mid_cap': 0.001,      # 中型股 0.1%
            'small_cap': 0.002,    # 小型股 0.2%
        }
        
    def calculate_total_cost(self, symbol, quantity, price, market_cap='large_cap'):
        """
        計算總交易成本
        
        成本 = 佣金 + 滑點 + 印花稅(如適用)
        """
        trade_value = quantity * price
        
        # 佣金
        if symbol.endswith('.HK'):
            commission = trade_value * self.commission_rate['HK_STOCK']
            stamp_duty = trade_value * 0.001  # HK 印花稅
        else:
            commission = trade_value * self.commission_rate['US_STOCK']
            stamp_duty = 0
            
        # 滑點
        slippage = trade_value * self.slippage_model.get(market_cap, 0.001)
        
        # 總成本
        total_cost = commission + slippage + stamp_duty
        
        # 成本比率
        cost_rate = total_cost / trade_value
        
        return {
            'commission': commission,
            'slippage': slippage,
            'stamp_duty': stamp_duty,
            'total_cost': total_cost,
            'cost_rate': cost_rate,
            'effective_price': price * (1 + cost_rate)  # 買入時
        }
```

### 3.1 風險度量系統

```python
class RiskMetrics:
    """
    進階風險度量
    根據書籍第六章：VaR/CVaR/最大回撤
    """
    
    def __init__(self, confidence_level=0.95):
        self.confidence_level = confidence_level
        
    def calculate_var(self, returns, portfolio_value):
        """
        VaR (Value at Risk)
        在 confidence_level 下的最大損失
        
        例如: VaR 95% = $1000
        → 95% 概率下，最大損失不超過 $1000
        """
        import numpy as np
        returns = np.array(returns)
        
        # 歷史模擬法
        var_percentile = (1 - self.confidence_level) * 100
        var_return = np.percentile(returns, var_percentile)
        
        var_dollar = portfolio_value * abs(var_return)
        
        return var_dollar
    
    def calculate_cvar(self, returns, portfolio_value):
        """
        CVaR (Conditional VaR / Expected Shortfall)
        超過 VaR 的平均損失
        
        即: 當損失超過 VaR 時，平均損失是多少
        """
        import numpy as np
        returns = np.array(returns)
        
        var_percentile = (1 - self.confidence_level) * 100
        var_return = np.percentile(returns, var_percentile)
        
        # 超過 VaR 的所有損失的平均值
        cvar_return = returns[returns <= var_return].mean()
        
        cvar_dollar = portfolio_value * abs(cvar_return)
        
        return cvar_dollar
    
    def calculate_max_drawdown(self, equity_curve):
        """
        最大回撤 (Maximum Drawdown)
        從峰值到谷底的最大跌幅
        """
        import numpy as np
        equity = np.array(equity_curve)
        
        # 計算峰值
        running_max = np.maximum.accumulate(equity)
        
        # 計算回撤
        drawdown = (equity - running_max) / running_max
        
        # 最大回撤
        max_dd = drawdown.min()
        
        return max_dd
```

### 3.2 風險控制規則

```
【風險控制規則】
- VaR 報警: 當日損失 > VaR 95% → 暫停交易
- CVaR 報警: 損失 > CVaR 95% → 全面平倉
- 最大回撤: > 15% → 停止新交易
- 最大回撤: > 20% → 進入防守模式 (只平倉不開倉)
- 最大回撤: > 30% → 策略暫停，檢討
```

### 4.2 績效評估指標

```python
class PerformanceMetrics:
    """
    策略績效評估
    根據書籍建議的關鍵指標
    """
    
    def calculate_all_metrics(self, returns, benchmark_returns=None, risk_free_rate=0.04):
        """
        計算所有績效指標
        """
        import numpy as np
        
        returns = np.array(returns)
        
        # 1. 夏普比率 (Sharpe Ratio)
        excess_returns = returns - risk_free_rate / 252
        sharpe = np.mean(excess_returns) / np.std(excess_returns) * np.sqrt(252)
        
        # 2. 資訊比率 (Information Ratio) - 如果有 benchmark
        if benchmark_returns is not None:
            active_returns = returns - benchmark_returns
            info_ratio = np.mean(active_returns) / np.std(active_returns) * np.sqrt(252)
        else:
            info_ratio = None
            
        # 3. 卡爾瑪比率 (Calmar Ratio)
        max_dd = self.calculate_max_drawdown(returns)
        annual_return = np.mean(returns) * 252
        calmar = annual_return / abs(max_dd) if max_dd != 0 else 0
        
        # 4. 索提諾比率 (Sortino Ratio) - 只看下行波動
        downside_returns = returns[returns < 0]
        sortino = np.mean(excess_returns) / np.std(downside_returns) * np.sqrt(252)
        
        # 5. 勝率 (Win Rate)
        win_rate = len(returns[returns > 0]) / len(returns)
        
        # 6. 平均贏/輸比率
        avg_win = np.mean(returns[returns > 0]) if len(returns[returns > 0]) > 0 else 0
        avg_loss = abs(np.mean(returns[returns < 0])) if len(returns[returns < 0]) > 0 else 0
        
        return {
            'sharpe_ratio': sharpe,
            'information_ratio': info_ratio,
            'calmar_ratio': calmar,
            'sortino_ratio': sortino,
            'win_rate': win_rate,
            'avg_win': avg_win,
            'avg_loss': avg_loss,
            'profit_factor': avg_win / avg_loss if avg_loss != 0 else 0,
            'max_drawdown': max_dd,
        }
```

### 5.2 市場狀態識別

```python
class MarketRegimeDetector:
    """
    市場狀態識別
    根據書籍第七章：不同策略適用於不同市場狀態
    """
    
    def detect_regime(self, prices, lookback=20):
        """
        識別當前市場狀態
        
        返回:
        - 'trending_up': 上升趨勢
        - 'trending_down': 下降趨勢
        - 'sideways': 震盪市
        """
        import numpy as np
        
        returns = np.diff(prices[-lookback:]) / prices[-lookback:-1]
        
        # 趨勢強度
        trend_strength = abs(np.mean(returns)) / np.std(returns)
        
        # 波動性
        volatility = np.std(returns)
        
        if trend_strength > 1.5:
            if np.mean(returns) > 0:
                return 'trending_up'
            else:
                return 'trending_down'
        else:
            return 'sideways'
    
    def select_strategy(self, regime):
        """
        根據市場狀態選擇策略
        """
        strategy_map = {
            'trending_up': ['momentum', 'breakout'],
            'trending_down': ['mean_reversion', 'momentum_short'],
            'sideways': ['mean_reversion', 'pair_trading'],
        }
        
        return strategy_map.get(regime, ['momentum'])
```
